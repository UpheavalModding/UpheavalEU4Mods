### Get Country-specific Abilities Unlocked
defined_text = {
    name = GetCountrySpecificAbilitiesUnlocked

	# Vanilla Settings
	text = {
        localisation_key = age_overhaul_country_specific_abilities_default
        trigger = {
			has_global_flag = age_overhaul_option_country_specific_abilities_special_country_shortcuts_enabled
			NOT = {
				has_global_flag = age_overhaul_option_country_specific_abilities_disabled
				has_global_flag = age_overhaul_option_country_specific_abilities_special_requirements_enabled
				has_global_flag = age_overhaul_option_country_specific_abilities_unlocked_for_all
			}
        }
	}
	# Country-Specific Abilities Disabled
	text = {
        localisation_key = age_overhaul_country_specific_abilities_disabled
        trigger = {
			has_global_flag = age_overhaul_option_country_specific_abilities_disabled
			NOT = {
				has_global_flag = age_overhaul_option_country_specific_abilities_special_requirements_enabled
				has_global_flag = age_overhaul_option_country_specific_abilities_special_country_shortcuts_enabled
				has_global_flag = age_overhaul_option_country_specific_abilities_unlocked_for_all
			}
        }
	}
	# Country-Specific Abilities Unlocked for All, No Special Requirements
	text = {
        localisation_key = age_overhaul_country_specific_abilities_unlocked
        trigger = {
			has_global_flag = age_overhaul_option_country_specific_abilities_unlocked_for_all
			OR = {
				has_global_flag = age_overhaul_option_country_specific_abilities_special_country_shortcuts_enabled
				NOT = { has_global_flag = age_overhaul_option_country_specific_abilities_special_country_shortcuts_enabled }
			}
			NOT = {
				has_global_flag = age_overhaul_option_country_specific_abilities_disabled
				has_global_flag = age_overhaul_option_country_specific_abilities_special_requirements_enabled
			}
        }
	}
	# Country-Specific Abilities Unlocked for All, Special Requirements. No Shortcuts
	text = {
        localisation_key = age_overhaul_country_specific_abilities_no_unique_countries
        trigger = {
			has_global_flag = age_overhaul_option_country_specific_abilities_unlocked_for_all
			has_global_flag = age_overhaul_option_country_specific_abilities_special_requirements_enabled
			NOT = {
				has_global_flag = age_overhaul_option_country_specific_abilities_special_country_shortcuts_enabled
				has_global_flag = age_overhaul_option_country_specific_abilities_disabled
			}
        }
	}
	# Country-Specific Abilities Unlocked for All, Special Requirements. Shortcuts
	text = {
        localisation_key = age_overhaul_country_specific_abilities_unlocked_no_unique_countries
        trigger = {
			has_global_flag = age_overhaul_option_country_specific_abilities_unlocked_for_all
			has_global_flag = age_overhaul_option_country_specific_abilities_special_requirements_enabled
			has_global_flag = age_overhaul_option_country_specific_abilities_special_country_shortcuts_enabled
			NOT = {
				has_global_flag = age_overhaul_option_country_specific_abilities_disabled
			}
        }
	}
}

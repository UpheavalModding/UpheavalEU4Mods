### Get Age of Industrialization Enabled
defined_text = {
    name = GetAgeOfIndustrializationEnabled

    text = {
        localisation_key = age_overhaul_age_of_industrialization_enabled
        trigger = {
			has_global_flag = age_overhaul_option_age_of_industrialization_enabled
        }
    }
	text = {
        localisation_key = age_overhaul_age_of_industrialization_disabled
        trigger = {
			NOT = { has_global_flag = age_overhaul_option_age_of_industrialization_enabled }
        }
	}
}

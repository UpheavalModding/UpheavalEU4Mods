namespace = age_overhaul_options_menu_carryover

# Decision Enablement Menu
country_event = {
	id = age_overhaul_options_menu_carryover.1
	title = age_overhaul_options_menu_carryover.1.title
	desc = age_overhaul_options_menu_desc
	picture = CITY_DEVELOPMENT_eventPicture

	is_triggered_only = yes

	# Back to Main Menu
	option = {
		name = age_overhaul_options_back
		country_event = { id = age_overhaul_menu_skipper.1 }
	}

	# Randomize Carrying Over Age Abilities
	option = {
		name = age_overhaul_options_random
		hidden_effect = {
			clr_global_flag = age_overhaul_option_carryover_enabled

			random_list = {
				10 = {
					set_global_flag = age_overhaul_option_carryover_enabled
				}
				10 = {
					clr_global_flag = age_overhaul_option_carryover_enabled
				}
			}
			country_event = { id = age_overhaul_menu_skipper.1 }
		}
	}

	# Enable Carrying Over Age Abilities
	option = {
		name = age_overhaul_options_menu_carryover.1.a

		trigger = {
			NOT = { has_global_flag = age_overhaul_option_carryover_enabled }
		}

		hidden_effect = {
			set_global_flag = age_overhaul_option_carryover_enabled
			country_event = { id = age_overhaul_menu_skipper.1 }
		}
	}
	# Disable Carrying Over Age Abilities
	option = {
		name = age_overhaul_options_menu_carryover.1.b

		trigger = {
			has_global_flag = age_overhaul_option_carryover_enabled
		}

		hidden_effect = {
			clr_global_flag = age_overhaul_option_carryover_enabled
			country_event = { id = age_overhaul_menu_skipper.1 }
		}
	}
}
